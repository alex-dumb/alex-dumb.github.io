    <script>
        function showPlanet(planet) {
            const planetImage = document.getElementById('planet-image');
            const planetDescription = document.getElementById('planet-description');

            // Update image source and description based on the selected planet
            switch (planet) {
                case 'Mercury':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/mercury-truecolor-2008.jpg';
                    planetImage.alt = 'Image of Mercury';
                    planetDescription.textContent = 'Mercury is the smallest planet in our solar system and closest to the Sun.';
                    break;
                case 'Venus':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/venus-20090324-1500c.jpg';
                    planetImage.alt = 'Image of Venus';
                    planetDescription.textContent = 'Venus is the second planet from the Sun and is often called Earth\'s "sister planet".';
                    break;
                case 'Earth':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/epic-earth-aug21-2017.png';
                    planetImage.alt = 'Image of Earth';
                    planetDescription.textContent = 'Earth is the third planet from the Sun and the only known planet to support life.';
                    break;
                case 'Mars':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/pia21723-16.jpg';
                    planetImage.alt = 'Image of Mars';
                    planetDescription.textContent = 'Mars is often called the "Red Planet" due to its reddish appearance.';
                    break;
                case 'Jupiter':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/pia22957-16.jpg';
                    planetImage.alt = 'Image of Jupiter';
                    planetDescription.textContent = 'Jupiter is the largest planet in our solar system and is known for its Great Red Spot.';
                    break;
                case 'Saturn':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/pia23645-1041.jpg';
                    planetImage.alt = 'Image of Saturn';
                    planetDescription.textContent = 'Saturn is known for its spectacular ring system, made mostly of ice particles.';
                    break;
                case 'Uranus':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/pia18182.jpg';
                    planetImage.alt = 'Image of Uranus';
                    planetDescription.textContent = 'Uranus is the seventh planet from the Sun and is unique because it rotates on its side.';
                    break;
                case 'Neptune':
                    planetImage.src = 'https://www.nasa.gov/sites/default/files/thumbnails/image/neptune-pia01492.jpg';
                    planetImage.alt = 'Image of Neptune';
                    planetDescription.textContent = 'Neptune is the eighth and farthest known planet from the Sun in the Solar System.';
                    break;
                default:
                    planetImage.src = 'placeholder.png';
                    planetImage.alt = '';
                    planetDescription.textContent = 'Select a planet to learn more about it.';
            }
        }
    </script>
